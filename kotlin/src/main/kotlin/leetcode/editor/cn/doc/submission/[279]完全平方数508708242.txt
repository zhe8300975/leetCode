class Solution {
    fun numSquares(n: Int): Int {
        var dp=IntArray(n+1) { i: Int -> i }
        for(i in 1..n){
            var n=1
            while(i-n*n>=0){
                dp[i]=Math.min(dp[i-n*n]+1,dp[i])
                n++
            }
        }
        return dp[n]
    }
}
