    class Solution {
        fun numSquares(n: Int): Int {
            var dp = IntArray(n + 1) { i -> i }
            for(i in 1..n){
                var num=1
                while(i-num*num>=0){
                    dp[i]=Math.min(dp[i],dp[i-num*num]+1)
                    num++
                }
            }
            return dp[n]
        }
    }
