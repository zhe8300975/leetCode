class Solution {
    fun numSquares(n: Int): Int {
        val dp=IntArray(n+1)
        dp[0]=0
        dp[1]=1
        for(i in 2..n){
            dp[i]=i
           var j=0
            while(i-j*j>=0){
                dp[i]=Math.min(dp[i],dp[i-j*j]+1)
                j++
            }
        }
        return dp[n]

    }
}
