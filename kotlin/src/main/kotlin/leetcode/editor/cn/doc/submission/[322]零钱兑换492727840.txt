class Solution {
    fun coinChange(coins: IntArray, amount: Int): Int {
        val dp=IntArray(amount+1)
        for(i in 1..amount){
            dp[i]=-1
            coins.forEach {
                if(dp.getOrElse(i-it) { -1 } !=-1){
                    dp[i]=if(dp[i]==-1) dp[i-it]+1 else Math.min(dp[i],dp[i-it]+1)
                }
            }
        }
        return dp[amount]

    }
}
