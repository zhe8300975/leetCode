class Solution {
    fun canJump(nums: IntArray): Boolean {
        var i=0
        var max=0
        while(i<=max){
            if(i==nums.size-1){
                return true
            }
            max=Math.max(max,nums[i]+i)
            i++
        }
        return false
    }
}
